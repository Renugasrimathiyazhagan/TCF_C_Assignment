#include <stdio.h>

int main() {
    int i;

    // if else:
    int number = 5;
    if (number % 2 == 0)
        printf("Even\n");
    else
        printf("Odd\n");

    // For loop example
    printf("Counting from 1 to 5:\n");
    for (i = 1; i <= 5; i++) {
        printf("%d ", i);
    }

    return 0;
}
