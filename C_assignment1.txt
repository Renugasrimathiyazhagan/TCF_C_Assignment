#include <stdio.h>
#include <string.h>

#define MAX_STUDENTS 100
#define NUM_SUBJECTS 5

// Define the Student structure
struct Student {
    int id;
    char name[50];
    int marks[NUM_SUBJECTS];
    float average;
};

// Function to input student details
void inputStudents(struct Student s[], int n) {
    for (int i = 0; i < n; i++) {
        printf("\nEnter details for Student %d:\n", i + 1);
        printf("ID: ");
        scanf("%d", &s[i].id);
        getchar(); // Clear newline after ID input
        printf("Name: ");
        fgets(s[i].name, sizeof(s[i].name), stdin);
        s[i].name[strcspn(s[i].name, "\n")] = '\0'; // Remove newline
        printf("Enter marks for 5 subjects:\n");
        for (int j = 0; j < NUM_SUBJECTS; j++) {
            printf("Subject %d: ", j + 1);
            scanf("%d", &s[i].marks[j]);
        }
    }
}

// Function to calculate average marks for a student
void calculateAverage(struct Student *s) {
    int total = 0;
    for (int i = 0; i < NUM_SUBJECTS; i++) {
        total += s->marks[i];
    }
    s->average = total / (float)NUM_SUBJECTS;
}

// Function to display the topper student
void displayTopper(struct Student s[], int n) {
    int topperIndex = 0;
    for (int i = 1; i < n; i++) {
        if (s[i].average > s[topperIndex].average) {
            topperIndex = i;
        }
    }

    printf("\nTopper Student:\n");
    printf("ID: %d\n", s[topperIndex].id);
    printf("Name: %s\n", s[topperIndex].name);
    printf("Marks: ");
    for (int i = 0; i < NUM_SUBJECTS; i++) {
        printf("%d ", s[topperIndex].marks[i]);
    }
    printf("\nAverage: %.2f\n", s[topperIndex].average);
}

// Main function
int main() {
    struct Student students[5];
    int n = 5;

    // Input student details
    inputStudents(students, n);

    // Calculate average for each student
    for (int i = 0; i < n; i++) {
        calculateAverage(&students[i]);
    }

    // Display topper student
    displayTopper(students, n);

    return 0;
}
